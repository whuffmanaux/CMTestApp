	ROLE_USER = 0
	ROLE_ADMIN = 1

	class User(db.Model):
  	  id = db.Column(db.Integer, primary_key = True)
  	  name = db.Column(db.String(64), index = True, unique = True)
  	  role = db.Column(db.SmallInteger, default = ROLE_USER)

  	  def __repr__(self):
  	      return '<User %r>' % (self.name)
